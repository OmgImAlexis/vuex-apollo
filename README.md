## Install

```js
import Vue from 'vue'
import vuexApollo from 'vuex-apollo'
import App from './App.vue'

Vue.use(vuexApollo, {
  uri: '',
  modules: []
})

new Vue({
  el: '#app',
  render: h => h(App)
})
```
